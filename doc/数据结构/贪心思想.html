                                          for (char[] c : path) {
                                              Arrays.fill(c, '.');

                                          if (row == n) {

                                          }
                                  

                                          for (int col = 0; col < n; col++) {
                                              if (isValid(row, col, path, n)) {
                                                  path[row][col] = 'Q';
                                                  backtracking(row + 1, n, path);

                                          for (int i = 0; i < row; i++) {

                                                  return false;
                                              }

                                                  return false;
                                              }

                                              if (path[i][j] == 'Q') {

                                          int end = 0;
                                          int maxPosition = 0;
